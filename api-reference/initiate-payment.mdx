---
title: 'Initiate Payment'
api: 'POST https://api.rolla.io/api/v1/transaction/initiate'
description: 'Create a new payment transaction'
---

## Authentication

All requests must include your secret API key in the Authorization header:

```
Authorization: Bearer sk_live_your_secret_key
```

<Note>
  Use `sk_test_*` keys for testing and `sk_live_*` keys for production.
</Note>

## Request Body

<ParamField body="amount" type="number" required>
  The payment amount (e.g., 1000 for 1000 units of the currency)
</ParamField>

<ParamField body="currency" type="string" required>
  Three-letter ISO currency code (e.g., "USD", "NGN", "GBP")
</ParamField>

<ParamField body="channel" type="string" required>
  Payment channel: `card`, `bank_transfer`, or `ussd`
</ParamField>

<ParamField body="customer" type="object" required>
  Customer information
  <Expandable title="properties">
    <ParamField body="email" type="string" required>
      Customer's email address
    </ParamField>
    <ParamField body="name" type="string">
      Customer's full name
    </ParamField>
    <ParamField body="phone" type="string">
      Customer's phone number
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="card" type="object">
  Card details (required when channel is `card`)
  <Expandable title="properties">
    <ParamField body="number" type="string" required>
      Card number (13-19 digits)
    </ParamField>
    <ParamField body="expiryMonth" type="string" required>
      Card expiry month (2 digits, e.g., "01" for January)
    </ParamField>
    <ParamField body="expiryYear" type="string" required>
      Card expiry year (2 or 4 digits, e.g., "27" or "2027")
    </ParamField>
    <ParamField body="cvv" type="string" required>
      Card CVV/CVC (3-4 digits)
    </ParamField>
    <ParamField body="holderName" type="string">
      Cardholder name as it appears on the card
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="bankTransfer" type="object">
  Bank transfer details (required when channel is `bank_transfer`)
  <Expandable title="properties">
    <ParamField body="bankCode" type="string" required>
      Bank code for the transfer
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="billing" type="object">
  Billing address (recommended for international cards to reduce declines)
  <Expandable title="properties">
    <ParamField body="address" type="string">
      Street address
    </ParamField>
    <ParamField body="city" type="string">
      City
    </ParamField>
    <ParamField body="state" type="string">
      State or province
    </ParamField>
    <ParamField body="country" type="string">
      Country code (e.g., "US", "NG")
    </ParamField>
    <ParamField body="postalCode" type="string">
      Postal/ZIP code
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="callbackUrl" type="string">
  URL to redirect the customer after payment completion
</ParamField>

<ParamField body="description" type="string">
  Optional description for the payment (max 500 characters)
</ParamField>

<ParamField body="metadata" type="object">
  Additional key-value pairs to attach to the transaction (e.g., order ID)
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="status" type="number">
  HTTP status code (201 for success)
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="reference" type="string">
      Unique transaction reference (e.g., `TXN_gGbEEvK25eVgmgrf`)
    </ResponseField>
    <ResponseField name="status" type="string">
      Transaction status: `pending`, `requires_action`, `processing`, `successful`, `failed`
    </ResponseField>
    <ResponseField name="amount" type="number">
      Transaction amount
    </ResponseField>
    <ResponseField name="currency" type="string">
      Currency code
    </ResponseField>
    <ResponseField name="fee" type="object">
      Fee breakdown
      <Expandable title="properties">
        <ResponseField name="amount" type="number">
          Total fee amount
        </ResponseField>
        <ResponseField name="type" type="string">
          Fee type: `local` or `international`
        </ResponseField>
        <ResponseField name="breakdown" type="object">
          <ResponseField name="percentage" type="number">
            Percentage fee
          </ResponseField>
          <ResponseField name="flat" type="number">
            Flat fee amount
          </ResponseField>
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="netAmount" type="number">
      Amount after fees (amount - fee)
    </ResponseField>
    <ResponseField name="nextAction" type="object">
      Next action required (if status is `requires_action`)
      <Expandable title="properties">
        <ResponseField name="type" type="string">
          Action type: `redirect`, `pin`, `otp`, `avs`
        </ResponseField>
        <ResponseField name="redirectUrl" type="string">
          URL to redirect the customer (for 3D Secure)
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO 8601 timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

## Transaction Statuses

| Status | Description |
|--------|-------------|
| `pending` | Transaction has been created but not yet processed |
| `requires_action` | Additional action required (3DS, PIN, OTP) |
| `processing` | Transaction is being processed |
| `successful` | Transaction completed successfully |
| `failed` | Transaction failed |

## Next Steps

After initiating a payment:

1. If `status` is `requires_action`, redirect the customer to `nextAction.redirectUrl`
2. The customer completes authentication (3D Secure, PIN, etc.)
3. Customer is redirected back to your `callbackUrl`
4. Use the [Verify Transaction](/api-reference/verify-transaction) endpoint to confirm the final status
